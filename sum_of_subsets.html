int n, target;
int w[100], x[100];

void subset(int s, int k, int r) {
    x[k] = 1;
    if (s + w[k] == target) {
        for (int i = 0; i <= k; i++)
            if (x[i])
                printf("%d ", w[i]);
        printf("\n");
    } else if (s + w[k] + w[k + 1] <= target) {
        subset(s + w[k], k + 1, r - w[k]);
    }

    if ((s + r - w[k] >= target) && (s + w[k + 1] <= target)) {
        x[k] = 0;
        subset(s, k + 1, r - w[k]);
    }
}

int main() {
    int sum = 0;
    scanf("%d", &n);

    for (int i = 0; i < n; i++) {
        scanf("%d", &w[i]);
        sum += w[i];
    }

    scanf("%d", &target);

    time_t start, end;
    start = time(NULL);

    subset(0, 0, sum);

    end = time(NULL);

    printf("%.2f seconds\n", difftime(end, start));

    return 0;
}
