int bpm(int u, int graph[100][100], int seen[], int matchR[], int n) {
    for (int v = 0; v < n; v++) {
        if (graph[u][v] && !seen[v]) {
            seen[v] = 1;
            if (matchR[v] < 0 || bpm(matchR[v], graph, seen, matchR, n)) {
                matchR[v] = u;
                return 1;
            }
        }
    }
    return 0;
}

int maxBPM(int graph[100][100], int m, int n) {
    int matchR[100];
    memset(matchR, -1, sizeof(matchR));

    int result = 0;
    for (int u = 0; u < m; u++) {
        int seen[100] = {0};
        if (bpm(u, graph, seen, matchR, n))
            result++;
    }
    return result;
}

int main() {
    int m, n;
    scanf("%d %d", &m, &n);

    int graph[100][100];

    for (int i = 0; i < m; i++)
        for (int j = 0; j < n; j++)
            scanf("%d", &graph[i][j]);

    time_t start, end;
    start = time(NULL);

    int res = maxBPM(graph, m, n);

    end = time(NULL);

    printf("%d\n", res);
    printf("%.2f seconds\n", difftime(end, start));

    return 0;
}
