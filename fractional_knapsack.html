struct Item {
    int value, weight;
};

void swap(struct Item *a, struct Item *b) {
    struct Item temp = *a;
    *a = *b;
    *b = temp;
}

int partition(struct Item arr[], int low, int high) {
    float pivot = (float)arr[high].value / arr[high].weight;
    int i = low - 1;

    for (int j = low; j < high; j++) {
        float ratio = (float)arr[j].value / arr[j].weight;
        if (ratio > pivot) {
            i++;
            swap(&arr[i], &arr[j]);
        }
    }
    swap(&arr[i + 1], &arr[high]);
    return i + 1;
}

void quickSort(struct Item arr[], int low, int high) {
    if (low < high) {
        int pi = partition(arr, low, high);
        quickSort(arr, low, pi - 1);
        quickSort(arr, pi + 1, high);
    }
}

double fractionalKnapsack(int W, struct Item arr[], int n) {
    quickSort(arr, 0, n - 1);

    double total = 0.0;

    for (int i = 0; i < n; i++) {
        if (arr[i].weight <= W) {
            W -= arr[i].weight;
            total += arr[i].value;
        } else {
            total += (double)arr[i].value * W / arr[i].weight;
            break;
        }
    }
    return total;
}

int main() {
    int n, W;
    scanf("%d", &n);

    struct Item arr[n];

    for (int i = 0; i < n; i++)
        scanf("%d %d", &arr[i].value, &arr[i].weight);

    scanf("%d", &W);

    time_t start, end;
    start = time(NULL);

    double result = fractionalKnapsack(W, arr, n);

    end = time(NULL);

    printf("%.2f\n", result);
    printf("%.2f seconds\n", difftime(end, start));

    return 0;
}
