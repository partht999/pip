{

    {
        {1,0},
        {0,1},
        {1,1},
        {1,0}
    };

    int max[4][2] =
    {
        {2,1},
        {1,2},
        {1,2},
        {2,1}
    };

    int avail[2] = {1,2};

    int need[4][2];

    {

Q9Disk Scheduling
A cloud backup service receives disk I/O requests at tracks: 55, 58, 39, 18, 90, 160, 150, 38, 184. Head starts at 100 moving toward lower tracks. Apply C-SCAN algorithm and find total head movement.
●	Implement C-SCAN: head moves in one direction, jumps to opposite end, continues same direction
●	Print service order and mark where the head jumps
●	Calculate total head movement including the jump distance


STEP 1 — CREATE FILE
nano cscan.c
________________________________________
STEP 2 — WRITE CODE
#include 
#include 

int main()
{
    int req[9] =
    {
        55,58,39,18,90,
        160,150,38,184
    };

    int head = 100;

    int total = 0;

    int disk_size = 199;

    int temp;

    for(int i=0;i<9;i++)
    {
        for(int j=i+1;j<9;j++)
        {
            if(req[i] > req[j])
            {
                temp = req[i];
                req[i] = req[j];
                req[j] = temp;
            }
        }
    }

    int pos = 0;

    for(int i=0;i<9;i++)
    {
        if(req[i] > head)
        {
            pos = i;
            break;
        }
    }

    int current = head;

    printf("C-SCAN Service Order:\n");

    printf("%d ",head);

    for(int i=pos-1;i>=0;i--)
    {
        total += abs(current - req[i]);

        current = req[i];

        printf("-> %d ",current);
    }

    total += abs(current - 0);

    current = 0;

    printf("-> 0 ");

    total += abs(current - disk_size);

    current = disk_size;

    printf("-> JUMP -> %d ",current);

    for(int i=8;i>=pos;i--)
    {
        total += abs(current - req[i]);

        current = req[i];

        printf("-> %d ",current);
    }

    printf("\n\nTotal Head Movement = %d\n",
            total);

    return 0;
}

Q10Memory Address Translation
A web browser uses a two-level page table for memory isolation between tabs. Given a virtual address and a 2-level page table structure, compute the physical address step by step.
●	Virtual address is 32-bit: bits [31:22] = L1 index, bits [21:12] = L2 index, bits [11:0] = offset
●	Given virtual address 0x00403004: extract L1=1, L2=3, offset=4 manually, look up in provided tables
●	L1[1] → L2 table base = 5000; L2[3] → frame = 7; compute and print physical address = 7×4096 + 4

STEP 1 — CREATE FILE
nano twolevel.c
________________________________________
STEP 2 — WRITE CODE
#include 

int main()
{
    unsigned int va = 0x00403004;

    unsigned int l1_index;

    unsigned int l2_index;

    unsigned int offset;

    unsigned int l2_base = 5000;

    unsigned int frame = 7;

    unsigned int physical;

    l1_index = (va >> 22) & 0x3FF;

    l2_index = (va >> 12) & 0x3FF;

    offset = va & 0xFFF;

    physical = (frame * 4096) + offset;

    printf("Virtual Address = 0x%X\n",
            va);

    printf("L1 Index = %u\n",
            l1_index);

    printf("L2 Index = %u\n",
            l2_index);

    printf("Offset = %u\n",
            offset);

    printf("L1[%u] -> L2 Table Base = %u\n",
            l1_index,
            l2_base);

    printf("L2[%u] -> Frame = %u\n",
            l2_index,
            frame);

    printf("Physical Address = %u\n",
            physical);

    printf("Physical Address (Hex) = 0x%X\n",
            physical);

    return 0;
}
