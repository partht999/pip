void computeLPS(char pat[], int M, int lps[]) {
    int len = 0;
    lps[0] = 0;
    int i = 1;

    while (i < M) {
        if (pat[i] == pat[len]) {
            len++;
            lps[i] = len;
            i++;
        } else {
            if (len != 0)
                len = lps[len - 1];
            else {
                lps[i] = 0;
                i++;
            }
        }
    }
}

void KMP(char txt[], char pat[]) {
    int M = strlen(pat);
    int N = strlen(txt);

    int lps[M];
    computeLPS(pat, M, lps);

    int i = 0, j = 0;

    while (i < N) {
        if (pat[j] == txt[i]) {
            i++;
            j++;
        }

        if (j == M) {
            printf("%d ", i - j);
            j = lps[j - 1];
        } else if (i < N && pat[j] != txt[i]) {
            if (j != 0)
                j = lps[j - 1];
            else
                i++;
        }
    }
    printf("\n");
}

int main() {
    char txt[1000], pat[1000];

    scanf("%s", txt);
    scanf("%s", pat);

    time_t start, end;
    start = time(NULL);

    KMP(txt, pat);

    end = time(NULL);

    printf("%.2f seconds\n", difftime(end, start));

    return 0;
}
