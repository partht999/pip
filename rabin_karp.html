#define d 256
#define q 101

void rabinKarp(char txt[], char pat[]) {
    int M = strlen(pat);
    int N = strlen(txt);
    int i, j;
    int p = 0, t = 0, h = 1;

    for (i = 0; i < M - 1; i++)
        h = (h * d) % q;

    for (i = 0; i < M; i++) {
        p = (d * p + pat[i]) % q;
        t = (d * t + txt[i]) % q;
    }

    for (i = 0; i <= N - M; i++) {
        if (p == t) {
            for (j = 0; j < M; j++)
                if (txt[i + j] != pat[j])
                    break;
            if (j == M)
                printf("%d ", i);
        }

        if (i < N - M) {
            t = (d * (t - txt[i] * h) + txt[i + M]) % q;
            if (t < 0)
                t += q;
        }
    }
    printf("\n");
}

int main() {
    char txt[1000], pat[1000];

    scanf("%s", txt);
    scanf("%s", pat);

    time_t start, end;
    start = time(NULL);

    rabinKarp(txt, pat);

    end = time(NULL);

    printf("%.2f seconds\n", difftime(end, start));

    return 0;
}
