#define V 20

int g[V][V];

int main() {

    srand(time(0));

    // Step 1: Generate random undirected graph
    for (int i = 0; i < V; i++)
        for (int j = i + 1; j < V; j++)
            g[i][j] = g[j][i] = rand() % 2;

    printf("Graph adjacency matrix:\n");
    for (int i = 0; i < V; i++) {
        for (int j = 0; j < V; j++)
            printf("%d ", g[i][j]);
        printf("\n");
    }

    int cover[V] = {0};   // 1 if vertex is in vertex cover

    clock_t start = clock();

    // Step 2: Vertex Cover Approximation
    for (int i = 0; i < V; i++) {
        for (int j = i + 1; j < V; j++) {

            // if edge exists and both endpoints not yet removed
            if (g[i][j] == 1) {

                // add both endpoints to vertex cover
                cover[i] = 1;
                cover[j] = 1;

                // Step 3: REMOVE all edges incident on i and j
                for (int k = 0; k < V; k++) {
                    g[i][k] = 0;
                    g[k][i] = 0;

                    g[j][k] = 0;
                    g[k][j] = 0;
                }
            }
        }
    }

    clock_t end = clock();

    // Step 4: Print Vertex Cover
    printf("\nVertex Cover Set:\n");
    for (int i = 0; i < V; i++) {
        if (cover[i])
            printf("%d ", i);
    }

    printf("\nTime taken: %f seconds\n",
           (double)(end - start) / CLOCKS_PER_SEC);

    return 0;
}
