#define V 20   // at least 20 vertices

int g[V][V];

// Function to generate random undirected graph
void generateGraph(double density) {
    for (int i = 0; i < V; i++) {
        for (int j = 0; j < V; j++) {
            g[i][j] = 0;
        }
    }

    for (int i = 0; i < V; i++) {
        for (int j = i + 1; j < V; j++) {
            double r = (double)rand() / RAND_MAX;

            if (r < density) {
                g[i][j] = g[j][i] = 1;
            }
        }
    }
}

// 2-Approximation Vertex Cover
void vertexCoverApprox(int cover[]) {
    int i, j, k;

    for (i = 0; i < V; i++) {
        for (j = i + 1; j < V; j++) {

            if (g[i][j] == 1) {

                cover[i] = 1;
                cover[j] = 1;

                // remove all edges incident on i and j
                for (k = 0; k < V; k++) {
                    g[i][k] = 0;
                    g[k][i] = 0;

                    g[j][k] = 0;
                    g[k][j] = 0;
                }
            }
        }
    }
}

void printCover(int cover[]) {
    printf("Vertex Cover Set: ");
    for (int i = 0; i < V; i++) {
        if (cover[i])
            printf("%d ", i);
    }
    printf("\n");
}

int main() {
    srand(time(NULL));

    int cover[V] = {0};

    // generate random graph
    generateGraph(0.3); // 30% edge probability

    clock_t start = clock();

    vertexCoverApprox(cover);

    clock_t end = clock();

    printCover(cover);

    double time_taken = (double)(end - start) / CLOCKS_PER_SEC;

    printf("Execution Time: %f seconds\n", time_taken);

    return 0;
}
