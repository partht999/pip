// Simple restoring division for positive numbers only
int restoringDivision(int dividend, int divisor, int *remainder) {
    int quotient = 0;
    *remainder = 0;
    
    // Process 16 bits
    for (int i = 15; i >= 0; i--) {
        // Left shift remainder and quotient
        *remainder = (*remainder << 1);
        quotient = quotient << 1;
        
        // Get next bit from dividend
        int nextBit = (dividend >> i) & 1;
        *remainder = (*remainder & ~1) | nextBit;
        
        // Subtract divisor
        *remainder = *remainder - divisor;
        
        if (*remainder >= 0) {
            quotient = quotient | 1;  // Set quotient bit to 1
        } else {
            *remainder = *remainder + divisor;  // Restore
        }
    }
    
    return quotient;
}
